- 题2：反转链表

    ```go
    func reverseList(head *ListNode) *ListNode {
        if head == nil || head.Next == nil {return head}
        preNode := head
        curNode := head.Next
        preNode.Next = nil
        if curNode != nil {
            temp := curNode.Next
            curNode.Next = preNode
            preNode = curNode
            curNode = temp
        }
        return preNode
    }
    ```
